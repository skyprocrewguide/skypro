<script>

const modules = {

  module1: [

    'audio/module1_part1.mp3',

    'audio/module1_part2.mp3',

    'audio/module1_part3.mp3',

    'audio/module1_part4.mp3',

    'audio/module1_part5.mp3',

    'audio/module1_part6.mp3',

    'audio/module1_part7.mp3',

    'audio/module1_part8.mp3',

    'audio/module1_part9.mp3',

    'audio/module1_part10.mp3',

    'audio/module1_part11.mp3',

    'audio/module1_part12.mp3'

  ]

};


const currentPart = { module1: 0 };

const audioPlayer = document.getElementById('audioPlayer');

const partIndicator = document.getElementById('module1-part');


function playModulePart(moduleName, partIndex) {

  currentPart[moduleName] = partIndex;

  audioPlayer.src = modules[moduleName][partIndex];

  audioPlayer.play();

  updatePartIndicator(moduleName);

}


function nextPart(moduleName) {

  if(currentPart[moduleName] === undefined) currentPart[moduleName] = 0;

  const next = currentPart[moduleName] + 1;

  if(next < modules[moduleName].length){

    playModulePart(moduleName, next);

  } else {

    alert('Это последняя часть модуля');

  }

}


function prevPart(moduleName) {

  if(currentPart[moduleName] === undefined) currentPart[moduleName] = 0;

  const prev = currentPart[moduleName] - 1;

  if(prev >= 0){

    playModulePart(moduleName, prev);

  } else {

    alert('Это первая часть модуля');

  }

}


function updatePartIndicator(moduleName) {

  const indicator = document.getElementById(moduleName + '-part');

  indicator.textContent = `Часть ${currentPart[moduleName]+1} из ${modules[moduleName].length}`;

}
